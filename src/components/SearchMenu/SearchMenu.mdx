### Code:

#### Example State

```
}
```

#### Code Sample
```javascript
import SearchMenu from '@department-of-veterans-affairs/formation-react/SearchMenu'

<div style={{ backgroundColor: '#112e51', width: '100px' }}>
  <SearchMenu
    title='Default SearchMenu'
    id='default'
    isOpen={this.state.open}
    clickHandler={() => this.setState({ open: !this.state.open })}>
  </SearchMenu>
</div>
```
