### Code:

#### Date example
The date prop Object should look this:

```javascript
this.state = {
  date: {
    day: {
      value: '14',
      dirty: false,
    },
    month: {
      value: '4',
      dirty: false,
    },
    year: {
      value: '1983',
      dirty: false,
    },
  }
}
```

#### Example Code
```javascript
import Date from '@department-of-veterans-affairs/component-library/Date'

<Date
  onValueChange={date => this.setState({ date })}
  required={true}
  label='Please tell us a date'
  name='Date'
  date={this.state.date}
  requiredMessage='Please provide a response'
  invalidMessage='Please provide a valid date'
  ariaDescribedby='some-id'
/>;
```

          dirty: false,
        },

        },
      }

  render () {
    return (
      <div className="site-c-reactcomp__rendered">

      </div>
    );

}
